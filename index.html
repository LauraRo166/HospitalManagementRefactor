¿Qué tan limpio era el código?

LoginController.validate() — un método, demasiadas responsabilidades

⚠ ANTES — método monolítico
@Controller
public class LoginController {
    @Autowired LoginDao dao;
    @Autowired PatientPrescriptionDao dao1; // ¿para qué?
    @Autowired UsersInSystemDao dao2;       // ¿para qué?
    @Autowired LoginDao infoLog;

    public ModelAndView validate(String role,
        String username, String password,
        HttpServletRequest request) {

        // 1. Valida credenciales
        Login l1 = new Login(null, role, username, password);
        String userId = dao.validate(l1);

        // 2. Gestiona sesión HTTP
        HttpSession session = request.getSession();
        session.setAttribute("userInfo", l1);

        // 3. Log con contraseña en texto plano
        infoLog.logActivities("username=" + username
            + " password=" + password);

        // 4. Métricas del dashboard (recepcionista)
        int count = dao1.prescriptionPrintCount();

        // 5. Conteo de usuarios (admin)
        List users = dao2.getUsersInSystem();

        // 6. Decide qué vista renderizar
        mv.addObject("prescriptionsCount", count);
        mv.addObject("users_count", users.size());
        mv.setViewName("welcome");
        return mv;
    }
}
✓ DESPUÉS — responsabilidades separadas
@Controller
public class LoginController {
    @Autowired AuthenticationService authService;
    @Autowired DashboardService dashboardService;

    public ModelAndView login(String role,
        String username, String password,
        HttpServletRequest request) {

        LoginResult result = authService
            .authenticate(username, password, role);

        if (!result.isSuccess())
            return loginFailure();

        authService.createSession(request, result);

        Map<String,Object> model =
            dashboardService.buildFor(role);

        return welcomeView(model);
    }

    private ModelAndView welcomeView(
            Map<String,Object> model) {
        ModelAndView mv = new ModelAndView("welcome");
        model.forEach(mv::addObject);
        return mv;
    }

    private ModelAndView loginFailure() {
        ModelAndView mv = new ModelAndView("LoginView");
        mv.addObject("status", "false");
        return mv;
    }
}
